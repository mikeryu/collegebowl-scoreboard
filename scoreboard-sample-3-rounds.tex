\begin{game}

\begin{round}
\begin{tossup}
A pizza is cut into $8$ equal slices. If Maya eats $3$ slices, what fraction of the pizza does she eat?
\end{tossup}
\begin{tossupanswer}
$\frac{3}{8}$
\end{tossupanswer}
\begin{followup}
Solve: $7x-5=30$.
\end{followup}
\begin{followupanswer}
$x=5$
\end{followupanswer}
\begin{emceenotes}
Accept equivalent fraction notation for the toss-up answer.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
Round $2$: A school bus travels $24$ miles in the morning and $18$ miles in the afternoon. How many miles total does it travel?
\end{tossup}
\begin{tossupanswer}
$42$
\end{tossupanswer}
\begin{followup}
Find the area of a rectangle with length $12$ cm and width $5$ cm.
\end{followup}
\begin{followupanswer}
$60\text{ cm}^2$
\end{followupanswer}
\begin{emceenotes}
Require units for full credit on follow-up.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
The ratio of red marbles to blue marbles is $3:5$. If there are $15$ red marbles, how many blue marbles are there?
\end{tossup}
\begin{tossupanswer}
$25$
\end{tossupanswer}
\begin{followup}
Round $3$ follow-up: The temperature was $72^\circ$F at noon and dropped by $14^\circ$F by evening. What was the evening temperature?
\end{followup}
\begin{followupanswer}
$58^\circ\text{F}$
\end{followupanswer}
\begin{emceenotes}
Allow numerical answer without degree symbol if clearly identified as Fahrenheit.
\end{emceenotes}
\end{round}

\end{game}
