\vspace{0.75in}

{\huge\bf Toss-Up Question}




{\Large The number $2025$ is a perfect square. If an integer is chosen between $1$ and $2025$, inclusive, what is the probability it will also be a perfect square?}

\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{1}{45}$.}

\vspace{2in}

{\huge\bf Follow-up}

{\Large Two six-sided dice with sides labeled one through six are rolled. What is the probability that the sum of the numbers rolled is either a perfect square or a multiple of $4$?}

\vspace{0.3in}


\vspace{0.4in}

{\huge\bf Answer}

{\Large $\frac{13}{36}$.}

\medskip



\clearpage

{\huge\bf Toss-Up Question}

{\Large Find the largest value of $x$ at which the function $f(x)=\frac{x^2-x-2}{x^2+x-6}$ has a vertical asymptote.}\\
   

\vspace{0.5in}

{\huge\bf Answer}

{\Large $x=-3$.}

\vspace{2in}

{\huge\bf Follow-Up Question}

{\Large Find the coefficient of the cubic term of $(2x-1)^8$.} \ \\

\textbf{Note to the emcee:}
Read ``Find the coefficient of the cubic term of the quantity $2x-1$ raised to the eighth power.''


\vspace{0.5in}

{\huge\bf Answer}

{\Large $-448$.}

\clearpage

{\huge\bf Follow-Up Question}

\vspace{0.5in}

{\huge\bf Answer}

\vspace{0.75in}

{\huge\bf Toss-Up Question}

{\Large Let $p(x)=ax^3+bx^2+cx+d$. If $3i$ and $1$ are two roots of $p(x)$, find $a$, $b$, $c$, and $d$. }

\vspace{0.5in}



{\huge\bf Answer}

{\Large $a=1, b=-1, c=9, d=-9$.}

\vspace{2in}

{\huge\bf Follow-Up Question}

{\Large For what values of $b$ will the function $f(x)=2x^3+bx^2-4x^2+3x-2bx-6$ have only one real root?}



\vspace{0.5in}

{\huge\bf Answer}

{\Large $-2\sqrt{6}<b<2\sqrt{6}$}

\clearpage
