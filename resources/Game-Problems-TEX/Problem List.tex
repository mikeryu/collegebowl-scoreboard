\subsection{Logarithms}

\begin{enumerate}



\item 
\begin{enumerate}
    \item Write the expression $4\ln 3+2 \ln 5$ as a single natural logarithm without exponents.
    \item Simplify completely:
    $$
    \frac{1}{\log_{32} 10}+\frac{5}{\log_5 10}.
    $$
\end{enumerate}
\textbf{Answers: $\ln 2025$; $5$}\ \\


\item 
\begin{enumerate}
    \item Over a period of five years, the population of moose in a certain state increased by a factor of five. What was the population's constant annual rate of increase over this period?
    \item The value of a certain car depreciates at a rate of 3\% each year. How long will it take for the car to lose half of its original value? 
\end{enumerate}
\textbf{Answers: $\sqrt[5]{5}-1\%$; $-\frac{\ln 2}{\ln.97}$ years OR $\frac{\ln .5}{\ln.97}$ years}\ \\
Note to Emcee: After reading the answer for part $(a)$ also read ``This is about $38\%$.''\ \\
Note to Emcee: After reading the answer for part $(b)$ also read ``This is almost 23 years.''




\item 
\begin{enumerate}
    \item Suppose $a$, $b$, and $c$ are positive integers such that $a<b<c$ and
    $$
    \log a + \log b + \log c = 1.
    $$
    Find $a$, $b$, and $c$.
    \item Find all solutions to $$1+\log_2(x^2-45)=3+\log_2 x.$$
\end{enumerate}
\textbf{Answers: $a=1, b=2, c=5$; $x=9$.}


\item 
\begin{enumerate}
    \item Let $f(x)=e^{2x}$ and $g(x)=a\ln x$. Find $a$ such that $g(f(x))=x$ for all $x$.
    \item Suppose $f(x)=\ln (x+4)-3$. Find the range of $f^{-1}(x)$.
\end{enumerate}
\textbf{Answers: $a=\frac{1}{2}$; $(-4,\infty)$}





\subsection{Trig Identities}

\item 
\begin{enumerate}
    \item If $x$ is in the first quadrant and $\sin x = \frac{4}{5}$, find $\sin 2x$.
    \item Find all solutions to the equation $\cos x = \cos 2x$ in the interval $[0,2\pi]$.
\end{enumerate}
\textbf{Answers: $\frac{24}{25}$; $0, \frac{2\pi}{3}$, $\frac{4\pi}{3}$, $2\pi$}\\



\item 
\begin{enumerate}
    \item Evaluate $1-\cos^2 x - \sin^2 x$.
    \item Find all solutions to the following equation in the interval $[0,\pi]$.
    $$
    \frac{\sec^2\theta - 1}{\sec^2 \theta}=\frac{1}{4}
    $$  
\end{enumerate}
\textbf{Answers: $0$; $\frac{\pi}{6}$, $\frac{5\pi}{6}$}\\





\item 
\begin{enumerate}
    \item Find an exact value of $\tan 105$\textdegree. 
    \item Let $\theta$ be the acute angle at which the lines $y=3x-2$ and $y=2x+1$ intersect. Find $\tan \theta$.
\end{enumerate}
\textbf{Answers:$-2-\sqrt{3}$; $\frac{1}{7}$}



\item 
\begin{enumerate}
    \item If $\cos x = \frac{3}{4}$, what is $\sec (-x)$?
    \item Solve $(1+\sin x)(1+\sin (-x))=\frac{1}{4}$ for $x$ in the interval $[0,\pi]$. 
\end{enumerate}
\textbf{Answers: $\frac{4}{3}$; $\frac{\pi}{3}$, $\frac{2\pi}{3}$}

\subsection{Conic Sections}

\item 
\begin{enumerate}
    \item What is the area of the circle given by the equation $x^2-4x+y^2+2y-20=0$?
    \item Find the product of the length of the major and minor axes of the ellipse given by the equation $9x^2-18x+4y^2+8y=23$.
\end{enumerate}
\textbf{Answers: $25\pi$; $24$}

\item 
\begin{enumerate}
    \item Recall that a parabola consists of all points equidistant from a point, called the focus, and a line, called the directrix. What is the equation of the directrix of the parabola $y=x^2$?
    \item  A searchlight is made in the shape of a parabola with the light source placed at the focus. If the light source is $2$ feet from the vertex and the depth of the searchlight is $4$ feet, what should the width of the opening be? 
\end{enumerate}
\textbf{Answers: $y=-\frac{1}{4}$; $8\sqrt{2}$}

\item  
\begin{enumerate}
    \item Find the area of the ellipse given by the equation $2x^2+y^2=10$.
    \item Find the equation of the ellipse with foci at $(1, 3)$ and $(9,3)$ and a minor axis of length $8$. 
\end{enumerate}
\textbf{Answers: $5\sqrt{2}\pi$; $\frac{(x-5)^2}{32}+\frac{(y-3)^2}{16}=1$}


\item 
\begin{enumerate}
    \item A boat follows a hyperbolic path with Island A and Island B as foci. At 10:00am the boat is 10 miles from Island A and $15$ miles from Island B. At 2:00pm the boat is $8$ miles away from Island A. What are all possible distances between the boat and Island B at 2:00pm?
    \item Find the standard form of a hyperbola symmetric about the $y$-axis with one vertex at $(45,0)$ and one focus at $(\sqrt{2061}, 0)$.
\end{enumerate}
\textbf{Answers: $13$ miles or $3$ miles; $\frac{x^2}{2025}-\frac{y^2}{36}=1$}




\subsection{Probability}


\item 
\begin{enumerate}
    \item What is the probability that a factor of $2025$ is also a multiple of $5$?
    \item Suppose that $x$ is a factor of 2025. Given that $x$ is a multiple of $5$, what is the probability $x$ is also a multiple of $25$?
\end{enumerate}
\textbf{Answers: $\frac{2}{3}$ OR $66\frac{2}{3}\%$ OR $.\bar{6}$; $\frac{1}{2}$}

\item 
\begin{enumerate}
    \item A committee of three is chosen at random from a class with $10$ students. How many distinct committees can be chosen?
    \item A committee of three is chosen at random from a class of $7$ boys and $3$ girls. What is the probability that the committee will have at least one girl?
\end{enumerate}
\textbf{Answers: $120$; $\frac{17}{24}$}





\item 
\begin{enumerate}
    \item Suppose that $x$ and $y$ are positive integers such that $x+y<10$. What is the probability that $x=8$?
    \item Suppose that $x^2+y^2<1$. What is the probability that $x+y>1$?
\end{enumerate}
\textbf{Answers: $\frac{1}{36}$; $\frac{\pi-2}{4\pi}$ OR $\frac{1}{4}-\frac{1}{2\pi}$}



\item 
\begin{enumerate}
    \item Alaska license plates consist of three letters followed by three digits. How many license plates are possible if numbers and letters may be repeated? You may leave exponents in your answer.
    \item What is the probability that an Alaska license plate chosen at random will have at least one duplicate number or letter? Simplify your answer fully.
\end{enumerate}
\textbf{Answers: $26^3\cdot 10^3$ OR $260^3$ OR $17,576,000$ OR $26^3\cdot 1000$; $\frac{61}{169}$}









\subsection{Calculus}
\item 
\begin{enumerate}
    \item Find the third derivative of $e^{2x}$.
    \item Let $f(x)=xe^{x^2}$. Find the slope of the line tangent to $f'(x)$ at $x=1$. (You may leave your answer in terms of $e$).
\end{enumerate}
\textbf{Answers: $8e^{2x}$; $10e$.}

\item 
\begin{enumerate}
    \item Two cars start at the same point. The first car drives North at a speed of $10$ miles per hour and the second drives East at a speed of $20$ miles per hour. What is the distance between the two cars after three hours?
    \item Two cars start at the same point. The first car drives North at a speed of $10$ miles per hour and the second drives East at a speed of $20$ miles per hour. At what rate is the distance between the two cars changing?
\end{enumerate}
\textbf{Answers: $30\sqrt{5}$ miles; $10\sqrt{5}$ miles per hour}


\item 
\begin{enumerate}
    \item Evaluate
    $$\lim_{x\to 4}\frac{\sin(\pi x)}{x-4}$$
    \item Evaluate $$\lim_{x\to\infty}(e^x+x)^\frac{1}{x}.$$
\end{enumerate}
\textbf{Answers: $\pi$; $e$}

\item 
\begin{enumerate}
    \item Find the $x$-coordinates of all local maxima of the function $g(x)=\frac{1}{3}x^3-\frac{3}{2}x^2+2x+5.$
    \item Find the $x$-coordinates of all local maxima of the function
$$
F(x)=\int_1^x e^t(t+1)(t-1)(t-3)^2dt.
$$
\end{enumerate}
\textbf{Answers: $x=1$; $x=-1$.}\ \\
Note to Emcee: Be sure to wait for the entire answer to the toss-up question. An answer of $x=1$, $x=2$ is incorrect!



\subsection{Rates}


\item 
\begin{enumerate}
    \item The slope of the line connecting the points $(1, 6)$ and $(5,x)$ is $\frac{3}{4}$. Find $x$. 
    \item A biker bikes at a rate of 10 miles per hour for ten minutes and then rides at a constant speed of $x$ miles per hour for the remainder of an hour. If the biker's average velocity over the hour is $20$ miles per hour, find $x$.
\end{enumerate}
\textbf{Answers: $x=9$; $x=22$ miles per hour}



\item 
\begin{enumerate}
    \item A sink can be filled by a faucet in $5$ minutes, but it takes $7$ minutes to drain a full sink. If the faucet is running and the drain is  open, how long will it take to fill the sink?
    \item If $4$ liters of a $20\%$ acid solution are mixed with $6$ liters of a $25\%$ acid solution, what is the concentration of the resulting solution?
\end{enumerate}
\textbf{Answers: $17.5$ minutes OR $\frac{35}{2}$ minutes; $23\%$}



\item 
\begin{enumerate}
    \item Worker $A$ can complete a certain job in $4$ hours. Worker $B$ can complete the same job in $6$ hours. How long will it take them to complete the job if they work together?
    \item Algy can paint a certain room in one hour less than Gebra. Together they can complete the job in one hour and $12$ minutes. How long would it take each of them to complete the job individually?
\end{enumerate}
\textbf{Answers: $2.4$ hours OR $2\frac{2}{5}$ hours OR $2$ hours and $24$ minutes; Two hours for Algy and $3$ hours for Gebra}

\item 
\begin{enumerate}
    \item A biker leaves at 8:00 a.m. traveling $10$ miles per hour. Using the same route, a driver leaves at 10:00 a.m. traveling at $60$ miles per hour. At what time will the driver catch up to the biker?
    \item Runner $A$ and Runner $B$ are running around a small circular track with radius $6$ meters. If Runner A runs at a rate of 10 degrees per seconds and Runner B runs at a rate of $12$ degrees per second (both measured from the center of the circle), then how much more distance in meters will Runner B have covered than Runner A after $10$ seconds?
\end{enumerate}
\textbf{Answers: 10:24 am; $\frac{2}{3}\pi$}





\subsection{Final Round}



\item 
\begin{enumerate}
    \item The number $2025$ is a perfect square. If an integer is chosen between $1$ and $2025$, inclusive, what is the probability it will also be a perfect square?
    \item Two six-sided dice with sides labeled one through six are rolled. What is the probability that the sum of the numbers rolled is either a perfect square or a multiple of $4$?
\end{enumerate}
\textbf{Answers: $\frac{1}{45}$; $\frac{13}{36}$}



\item 
\begin{enumerate}
    \item Find the largest value of $x$ at which the function $f(x)=\frac{x^2-x-2}{x^2+x-6}$ has a vertical asymptote.
    \item Find the coefficient of the cubic term of $(2x-1)^8$.
\end{enumerate}
\textbf{Answers: $x=-3$; $-448$}\ \\

Read ``Find the coefficient of the cubic term of the quantity $2x-1$ raised to the eighth power.''



\item
\begin{enumerate}
    \item Simplify $\log_23\cdot\log_34\cdot\log_48$.
    \item Find the zeros of $f(x)=\log_{25}x+\log_5(\sqrt{x}+1)$.
\end{enumerate}
\textbf{Answers: $3$; $\frac{3\pm\sqrt{5}}{2}$}

\item 
\begin{enumerate}
    \item Find the equations of the two vertical tangent lines to $(x-2)^2+(y+3)^2=25$.
    \item How many times do the graphs of $y=\sin x$ and $y=\frac{x}{8\pi}$ intersect?
\end{enumerate}
\textbf{Answers: $x=-3, x=7$; $15$ times}

\item 
\begin{enumerate}
    \item How many distinct digits can be the last (units) digit of a perfect square?
    \item The average age of a group of students and teachers is $22$. The students' average age is $15$. The teachers' average age is $36$. Find the ratio of students to teachers. 
\end{enumerate}
\textbf{Answers: $6$; $2:1$ OR $2$ to $1$}

\item 
\begin{enumerate}
    \item Let $p(x)=ax^3+bx^2+cx+d$. If $3i$ and $1$ are two roots of $p(x)$, find $a$, $b$, $c$, and $d$. 
    \item For what values of $b$ will the function $f(x)=2x^3+bx^2-4x^2+3x-2bx-6$ have only one real root?
\end{enumerate}
\textbf{Answers: $a=1, b=-1, c=9, d=-9$; $-2\sqrt{6}<b<2\sqrt{6}$.}\ \\


\item 
\begin{enumerate}
    \item A certain town has coins with 1 cent, 2 cent, and 5 cent values. What is the smallest number of cents which cannot be made using at most three of these coins?
    \item Circle $\mathcal{O}$ has radius $5\sqrt{3}$. Sector $A\mathcal{O}B$ has a central angle of measure $\frac{2\pi}{3}$. Find the length of segment $\overline{AB}$. 
\end{enumerate}
\textbf{Answers: $13$; $15$ units}

\item 
\begin{enumerate}
    \item Simplify $2\cos^2(\frac{\pi}{12})-1$.
    \item Simplify 
    $$
    \csc^2\left(\frac{\pi}{8}\right)\csc^2\left(\frac{3\pi}{8}\right).
    $$
\end{enumerate}
\textbf{Answers: $\frac{\sqrt{3}}{2}$; $8$}

\item 
\begin{enumerate}
    \item How many $2\times 2 \times 2$ cubes can be fit in an $8\times 8\times 8$ cube?
    \item How many counting numbers under $10,000$ have an odd number of factors?
\end{enumerate}
\textbf{Answers: $64$; $99$}





\item 
\begin{enumerate}
    \item Find the vertex of the graph determined by the equation $y=3-2|x-2|$.
    \item Find the values of $x$ for which the graph determined by the equation $y=3-2|x-2|$ is below the $x$-axis.
\end{enumerate}
\textbf{Answers: $(2,3)$; $x<\frac{1}{2}$ or $x>\frac{7}{2}$ OR $x<.5$ or $x>3.5$.}\ \\

Note to the emcee: Read ``$y$ equals $3$ minus $2$ times the absolute value of the quantity $x-2$''



\end{enumerate}
