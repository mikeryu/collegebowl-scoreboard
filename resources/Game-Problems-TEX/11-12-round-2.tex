%-------------------Question/Follow-up 1-------------------------



{\huge\bf Toss-Up Question}



{\Large If $x$ is in the first quadrant and $\sin x = \frac{4}{5}$, find $\sin 2x$.}

\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{24}{25}$.}

\vspace{1.8in}

{\huge\bf Follow-Up Question}

{\Large Find all solutions to the equation $\cos x = \cos 2x$ in the interval $[0,2\pi]$.}





\vspace{0.25in}

{\huge\bf Answer}

{\Large $0, \frac{2\pi}{3}$, $\frac{4\pi}{3}$, $2\pi$.}

\clearpage

\vspace{0.75in}


  

%-------------------Question/Follow-up 2-------------------------



{\huge\bf Toss-Up Question}



{\Large Evaluate $1-\cos^2 x - \sin^2 x$.}

\vspace{0.5in}

{\huge\bf Answer}

{\Large $0$.}

\vspace{2in}

{\huge\bf Follow-Up Question}

{\Large Find all solutions to the following equation in the interval $[0,\pi]$.
    $$
    \frac{\sec^2\theta - 1}{\sec^2 \theta}=\frac{1}{4}
    $$ }\\



\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{\pi}{6}$, $\frac{5\pi}{6}$.}

\clearpage

\vspace{0.75in}




%-------------------Question/Follow-up 3-------------------------



{\huge\bf Toss-Up Question}



{\Large Find an exact value of $\tan 105$\textdegree.}

\vspace{0.5in}

{\huge\bf Answer}

{\Large $-2-\sqrt{3}$.}

\vspace{2in}

{\huge\bf Follow-Up Question}

{\Large Let $\theta$ be the acute angle at which the lines $y=3x-2$ and $y=2x+1$ intersect. Find $\tan \theta$.}

\medskip



\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{1}{7}$.}

\clearpage

\vspace{0.75in}



%-------------------Question/Follow-up 4-------------------------


{\huge\bf Toss-Up Question}




{\Large If $\cos x = \frac{3}{4}$, what is $\sec (-x)$?}

\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{4}{3}$.}

\vspace{2in}

{\huge\bf Follow-Up Question}

{\Large Solve $(1+\sin x)(1+\sin (-x))=\frac{1}{4}$ for $x$ in the interval $[0,\pi]$.}



\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{\pi}{3}$, $\frac{2\pi}{3}$.}
