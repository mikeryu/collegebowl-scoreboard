%-------------------Question/Follow-up 1-------------------------


{\huge\bf Toss-Up Question}




{\Large What is the probability that a factor of $2025$ is also a multiple of $5$?}

\vspace{0.3in}



{\huge\bf Answer}

{\Large $\frac{2}{3}$ OR $66\frac{2}{3}\%$ OR $.\bar{6}$.}\\


\vspace{1.5in}

{\huge\bf Follow-Up Question}

{\Large Suppose that $x$ is a factor of 2025. Given that $x$ is a multiple of $5$, what is the probability $x$ is also a multiple of $25$?}





\vspace{0.3in}

{\huge\bf Answer}

{\Large $\frac{1}{2}$.}

%-------------------Question/Follow-up 2-------------------------


{\huge\bf Toss-Up Question}





{\Large A committee of three is chosen at random from a class with $10$ students. How many distinct committees can be chosen?}

\vspace{0.5in}

{\huge\bf Answer}

{\Large $120$.}

\vspace{2in}

{\huge\bf Follow-Up Question}

{\Large A committee of three is chosen at random from a class of $7$ boys and $3$ girls. What is the probability that the committee will have at least one girl?}

\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{17}{24}$.}

\clearpage

\vspace{0.75in}




%-------------------Question/Follow-up 3-------------------------




{\huge\bf Toss-Up Question}



{\Large Suppose that $x$ and $y$ are positive integers such that $x+y<10$. What is the probability that $x=8$?}

\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{1}{36}$.}

\vspace{2in}

{\huge\bf Follow-Up Question}

{\Large Suppose that $x^2+y^2<1$. What is the probability that $x+y>1$?}

\medskip



\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{\pi-2}{4\pi}$ OR $\frac{1}{4}-\frac{1}{2\pi}$.}

\clearpage

\vspace{0.75in}



%-------------------Question/Follow-up 4-------------------------



{\huge\bf Toss-Up Question}



{\Large Alaska license plates consist of three letters followed by three digits. How many license plates are possible if numbers and letters may be repeated? You may leave exponents in your answer.}

\vspace{0.5in}

{\huge\bf Answer}

{\Large $26^3\cdot 10^3$ OR $260^3$ OR $17,576,000$ OR $26^3\cdot 1000$.}

\vspace{2in}

{\huge\bf Follow-Up Question}

{\Large What is the probability that an Alaska license plate chosen at random will have at least one duplicate number or letter? Simplify your answer fully.}



\vspace{0.5in}

{\huge\bf Answer}

{\Large $\frac{61}{169}$.}
