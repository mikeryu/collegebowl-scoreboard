\begin{game}

\begin{round}
\begin{tossup}
The number $2025$ is a perfect square. If an integer is chosen between $1$ and $2025$, inclusive, what is the probability it will also be a perfect square?
\end{tossup}
\begin{tossupanswer}
$\frac{1}{45}$.
\end{tossupanswer}
\begin{followup}
Two six-sided dice with sides labeled one through six are rolled. What is the probability that the sum of the numbers rolled is either a perfect square or a multiple of $4$?
\end{followup}
\begin{followupanswer}
$\frac{13}{36}$.
\end{followupanswer}
\begin{emceenotes}
Clarify that either condition qualifies: perfect square OR multiple of 4.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
Find the largest value of $x$ at which the function $f(x)=\frac{x^2-x-2}{x^2+x-6}$ has a vertical asymptote.
\end{tossup}
\begin{tossupanswer}
$x=-3$.
\end{tossupanswer}
\begin{followup}
Find the coefficient of the cubic term of $(2x-1)^8$.
\end{followup}
\begin{followupanswer}
$-448$.
\end{followupanswer}
\begin{emceenotes}
Read exponent carefully: quantity $2x-1$ raised to the 8th power.
Read aloud: ``Find the coefficient of the cubic term of the quantity $2x-1$ raised to the eighth power.''
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
Simplify $\log_23\cdot\log_34\cdot\log_48$.
\end{tossup}
\begin{tossupanswer}
$3$.
\end{tossupanswer}
\begin{followup}
Find the zeros of $f(x)=\log_{25}x+\log_5(\sqrt{x}+1)$.
\end{followup}
\begin{followupanswer}
$\frac{3\pm\sqrt{5}}{2}$.
\end{followupanswer}
\begin{emceenotes}
Accept either order of the two roots.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
Find the equations of the two vertical tangent lines to $(x-2)^2+(y+3)^2=25$.
\end{tossup}
\begin{tossupanswer}
$x=-3,\ x=7$.
\end{tossupanswer}
\begin{followup}
How many times do the graphs of $y=\sin x$ and $y=\frac{x}{8\pi}$ intersect?
\end{followup}
\begin{followupanswer}
$15$ times.
\end{followupanswer}
\begin{emceenotes}
Teams must provide the count, not an interval statement.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
How many distinct digits can be the last (units) digit of a perfect square?
\end{tossup}
\begin{tossupanswer}
$6$.
\end{tossupanswer}
\begin{followup}
The average age of a group of students and teachers is $22$. The students' average age is $15$. The teachers' average age is $36$. Find the ratio of students to teachers.
\end{followup}
\begin{followupanswer}
$2:1$ (or $2$ to $1$).
\end{followupanswer}
\begin{emceenotes}
Equivalent ratio forms are acceptable.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
Let $p(x)=ax^3+bx^2+cx+d$. If $3i$ and $1$ are two roots of $p(x)$, find $a$, $b$, $c$, and $d$.
\end{tossup}
\begin{tossupanswer}
$a=1,\ b=-1,\ c=9,\ d=-9$.
\end{tossupanswer}
\begin{followup}
For what values of $b$ will the function $f(x)=2x^3+bx^2-4x^2+3x-2bx-6$ have only one real root?
\end{followup}
\begin{followupanswer}
$-2\sqrt{6}<b<2\sqrt{6}$.
\end{followupanswer}
\begin{emceenotes}
Strict inequalities required; endpoints are not included.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
A certain town has coins with 1 cent, 2 cent, and 5 cent values. What is the smallest number of cents which cannot be made using at most three of these coins?
\end{tossup}
\begin{tossupanswer}
$13$.
\end{tossupanswer}
\begin{followup}
Circle $\mathcal{O}$ has radius $5\sqrt{3}$. Sector $A\mathcal{O}B$ has a central angle of measure $\frac{2\pi}{3}$. Find the length of segment $\overline{AB}$.
\end{followup}
\begin{followupanswer}
$15$ units.
\end{followupanswer}
\begin{emceenotes}
Unit labels optional.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
Simplify $2\cos^2\!\left(\frac{\pi}{12}\right)-1$.
\end{tossup}
\begin{tossupanswer}
$\frac{\sqrt{3}}{2}$.
\end{tossupanswer}
\begin{followup}
Simplify $\csc^2\!\left(\frac{\pi}{8}\right)\csc^2\!\left(\frac{3\pi}{8}\right)$.
\end{followup}
\begin{followupanswer}
$8$.
\end{followupanswer}
\begin{emceenotes}
No decimal approximation required.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
How many $2\times 2 \times 2$ cubes can be fit in an $8\times 8\times 8$ cube?
\end{tossup}
\begin{tossupanswer}
$64$.
\end{tossupanswer}
\begin{followup}
How many counting numbers under $10{,}000$ have an odd number of factors?
\end{followup}
\begin{followupanswer}
$99$.
\end{followupanswer}
\begin{emceenotes}
If needed, remind that odd number of factors implies perfect squares.
\end{emceenotes}
\end{round}

\begin{round}
\begin{tossup}
Find the vertex of the graph determined by the equation $y=3-2|x-2|$.
\end{tossup}
\begin{tossupanswer}
$(2,3)$.
\end{tossupanswer}
\begin{followup}
Find the values of $x$ for which the graph determined by the equation $y=3-2|x-2|$ is below the $x$-axis.
\end{followup}
\begin{followupanswer}
$x<\frac{1}{2}$ or $x>\frac{7}{2}$ (equivalently, $x<0.5$ or $x>3.5$). Both pieces required.
\end{followupanswer}
\begin{emceenotes}
Both intervals must be given for full credit.
For both toss-up and follow-up, read aloud: ``$y$ equals $3$ minus $2$ times the absolute value of the quantity $x-2$''.
\end{emceenotes}
\end{round}

\end{game}
